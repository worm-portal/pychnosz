def water_activity(composition: Dict[str, float], T: float = 298.15) -> float:
    """
    Calculate water activity from osmotic coefficient.
    
    Parameters
    ----------
    composition : dict
        Solution composition {species: molality}
    T : float, default 298.15
        Temperature in Kelvin
        
    Returns
    -------
    float
        Water activity
    """
    
    # Sum of all solute molalities
    m_total = sum(composition.values())
    
    if m_total <= 0:
        return 1.0
    
    # Osmotic coefficient
    phi = osmotic_coefficient(composition, T)
    
    # Water activity: ln(a_w) = -phi * M_w * sum(m_i)
    # where M_w = 0.018015 kg/mol (molar mass of water)
    M_w = 0.018015
    ln_aw = -phi * M_w * m_total
    
    return np.exp(ln_aw)

class ExtendedActivitySolver:
    """
    Extended activity coefficient solver with multiple models.
    
    This class provides a unified interface for different activity
    coefficient models and can select the appropriate model based
    on solution conditions.
    """
    
    def __init__(self):
        """Initialize the extended activity solver."""
        self.pitzer = PitzerModel()
        
        # Model selection criteria
        self.ionic_strength_limits = {
            'ideal': 0.001,
            'debye_huckel': 0.1,
            'bdot': 0.5,
            'pitzer': 6.0
        }
    
    def select_model(self, ionic_strength: float, composition: Dict[str, float]) -> str:
        """
        Select appropriate activity model based on conditions.
        
        Parameters
        ----------
        ionic_strength : float
            Solution ionic strength
        composition : dict
            Solution composition
            
        Returns
        -------
        str
            Recommended model name
        """
        
        if ionic_strength <= self.ionic_strength_limits['ideal']:
            return 'ideal'
        elif ionic_strength <= self.ionic_strength_limits['debye_huckel']:
            return 'debye_huckel'
        elif ionic_strength <= self.ionic_strength_limits['bdot']:
            return 'bdot'
        else:
            return 'pitzer'
    
    def calculate_activity_coefficients(self, composition: Dict[str, float],
                                      T: float = 298.15, P: float = 1.0,
                                      model: Optional[str] = None) -> Dict[str, float]:
        """
        Calculate activity coefficients for all species in solution.
        
        Parameters
        ----------
        composition : dict
            Solution composition {species: concentration}
        T : float, default 298.15
            Temperature in Kelvin
        P : float, default 1.0
            Pressure in bar
        model : str, optional
            Activity model to use. If None, auto-select based on conditions.
            
        Returns
        -------
        dict
            Activity coefficients {species: gamma}
        """
        
        # Calculate ionic strength
        ionic_strength = self._calculate_ionic_strength(composition)
        
        # Select model if not specified
        if model is None:
            model = self.select_model(ionic_strength, composition)
        
        # Calculate activity coefficients
        gamma = {}
        
        if model == 'ideal':
            for species in composition:
                gamma[species] = 1.0
        
        elif model == 'pitzer':
            for species in composition:
                if composition[species] > 0:
                    gamma[species] = self.pitzer.activity_coefficient(
                        species, composition, ionic_strength, T, P)
                else:
                    gamma[species] = 1.0
        
        else:
            # Use simplified models from equilibrium module
            from ..core.equilibrium import EquilibriumSolver
            solver = EquilibriumSolver()
            
            for species in composition:
                if composition[species] > 0:
                    if model == 'debye_huckel':
                        gamma[species] = solver._activity_debye_huckel(species, ionic_strength, T, P)
                    elif model == 'bdot':
                        gamma[species] = solver._activity_bdot(species, ionic_strength, T, P)
                    else:
                        gamma[species] = 1.0
                else:
                    gamma[species] = 1.0
        
        return gamma
    
    def _calculate_ionic_strength(self, composition: Dict[str, float]) -> float:
        """Calculate ionic strength from composition."""
        
        I = 0.0
        for species, conc in composition.items():
            if conc > 0:
                charge = self.pitzer._get_charge(species)
                I += 0.5 * conc * charge * charge
        
        return I

class PitzerModel:
    """
    Pitzer activity coefficient model for concentrated electrolyte solutions.
    
    The Pitzer model uses virial-type equations to calculate activity coefficients
    in solutions with ionic strengths up to 6 molal or higher.
    
    References:
    - Pitzer (1973) J. Phys. Chem. 77, 268-277
    - Pitzer & Mayorga (1973) J. Phys. Chem. 77, 2300-2308
    - Harvie et al. (1984) GCA 48, 723-751
    """
    
    def __init__(self):
        """Initialize the Pitzer model."""
        self.A_phi = 0.391  # Debye-Hückel parameter at 25°C (kg^0.5/mol^0.5)
        
        # Pitzer parameters database (simplified examples)
        self.parameters = {
            # Binary parameters: {cation-anion: [beta0, beta1, beta2, Cphi]}
            ('Na+', 'Cl-'): [0.0765, 0.2664, 0.0, -0.00127],
            ('Na+', 'SO4-2'): [0.01958, 1.113, -0.00114, 0.00497],
            ('K+', 'Cl-'): [0.04835, 0.2122, 0.0, -0.00084],
            ('Ca+2', 'Cl-'): [0.3159, 1.681, -0.00034, -0.00175],
            ('Mg+2', 'Cl-'): [0.35235, 1.681, -0.007, 0.00519],
            ('H+', 'Cl-'): [0.1775, 0.2945, 0.0, 0.0008],
            
            # Neutral species interactions
            ('CO2', 'Na+'): [0.1, 0.0],  # lambda parameters
            ('CO2', 'Cl-'): [-0.005, 0.0],
        }
        
        # Temperature dependence parameters (simplified)
        self.temp_params = {}
    
    def activity_coefficient(self, species: str, composition: Dict[str, float],
                           ionic_strength: float, T: float = 298.15,
                           P: float = 1.0) -> float:
        """
        Calculate activity coefficient using Pitzer model.
        
        Parameters
        ----------
        species : str
            Species name
        composition : dict
            Solution composition {species: molality}
        ionic_strength : float
            Solution ionic strength (mol/kg)
        T : float, default 298.15
            Temperature in Kelvin
        P : float, default 1.0
            Pressure in bar
            
        Returns
        -------
        float
            Activity coefficient
        """
        
        if ionic_strength <= 0:
            return 1.0
        
        # Get species charge
        charge = self._get_charge(species)
        
        if charge == 0:
            # Neutral species
            return self._neutral_activity_coefficient(species, composition, ionic_strength, T)
        else:
            # Ionic species
            return self._ionic_activity_coefficient(species, composition, ionic_strength, T, charge)
    
    def _ionic_activity_coefficient(self, species: str, composition: Dict[str, float],
                                  I: float, T: float, z: int) -> float:
        """Calculate activity coefficient for ionic species."""
        
        # Debye-Hückel term
        A_gamma = self.A_phi * 2.302585  # Convert to ln gamma
        sqrt_I = np.sqrt(I)
        
        ln_gamma_DH = -A_gamma * z*z * sqrt_I / (1 + 1.2 * sqrt_I)
        
        # Pitzer virial terms
        ln_gamma_virial = 0.0
        
        # Binary interactions
        for other_species, molality in composition.items():
            if other_species != species and molality > 0:
                other_charge = self._get_charge(other_species)
                
                # Cation-anion interactions
                if (z > 0 and other_charge < 0) or (z < 0 and other_charge > 0):
                    params = self._get_binary_params(species, other_species)
                    if params:
                        beta0, beta1, beta2, Cphi = params
                        
                        # Pitzer B and C functions
                        alpha1 = 2.0  # Standard value for 1-1, 2-1, 1-2 electrolytes
                        alpha2 = 50.0 # For 2-2 electrolytes (when beta2 != 0)
                        
                        g_alpha1 = 2 * (1 - (1 + alpha1 * sqrt_I) * np.exp(-alpha1 * sqrt_I)) / (alpha1**2 * I)
                        B_gamma = beta0 + beta1 * g_alpha1
                        
                        if beta2 != 0:
                            g_alpha2 = 2 * (1 - (1 + alpha2 * sqrt_I) * np.exp(-alpha2 * sqrt_I)) / (alpha2**2 * I)
                            B_gamma += beta2 * g_alpha2
                        
                        C_gamma = 3.0 * Cphi / (2 * np.sqrt(abs(z * other_charge)))
                        
                        ln_gamma_virial += molality * (2 * B_gamma + molality * C_gamma)
        
        # Triple interactions (simplified - usually small)
        # Would include theta and psi parameters for full model
        
        return np.exp(ln_gamma_DH + ln_gamma_virial)
    
    def _neutral_activity_coefficient(self, species: str, composition: Dict[str, float],
                                    I: float, T: float) -> float:
        """Calculate activity coefficient for neutral species."""
        
        # Neutral species interactions with ions (salting out/in)
        ln_gamma = 0.0
        
        for ion_species, molality in composition.items():
            if molality > 0:
                # Lambda interaction parameters
                lambda_param = self._get_lambda_param(species, ion_species)
                if lambda_param:
                    lambda_ni, xi_nii = lambda_param
                    ln_gamma += 2 * lambda_ni * molality + 3 * xi_nii * molality**2
        
        return np.exp(ln_gamma)
    
    def _get_charge(self, species: str) -> int:
        """Extract charge from species name."""
        
        if '+' in species:
            parts = species.split('+')
            if len(parts) > 1 and parts[-1].isdigit():
                return int(parts[-1])
            else:
                return 1
        elif '-' in species:
            parts = species.split('-')
            if len(parts) > 1 and parts[-1].isdigit():
                return -int(parts[-1])
            else:
                return -1
        else:
            return 0
    
    def _get_binary_params(self, species1: str, species2: str) -> Optional[List[float]]:
        """Get binary Pitzer parameters."""
        
        # Try both orders
        key1 = (species1, species2)
        key2 = (species2, species1)
        
        if key1 in self.parameters:
            return self.parameters[key1]
        elif key2 in self.parameters:
            return self.parameters[key2]
        else:
            # Return default parameters if not found
            return [0.0, 0.0, 0.0, 0.0]
    
    def _get_lambda_param(self, neutral: str, ion: str) -> Optional[Tuple[float, float]]:
        """Get lambda parameters for neutral-ion interactions."""
        
        key1 = (neutral, ion)
        key2 = (ion, neutral)
        
        # Simplified lookup - would be more comprehensive in full implementation
        if key1 in self.parameters:
            params = self.parameters[key1]
            return (params[0], params[1] if len(params) > 1 else 0.0)
        elif key2 in self.parameters:
            params = self.parameters[key2]
            return (params[0], params[1] if len(params) > 1 else 0.0)
        else:
            return (0.0, 0.0)
    
    def osmotic_coefficient(self, composition: Dict[str, float],
                          ionic_strength: float, T: float = 298.15) -> float:
        """
        Calculate osmotic coefficient using Pitzer model.
        
        Parameters
        ----------
        composition : dict
            Solution composition {species: molality}
        ionic_strength : float
            Solution ionic strength
        T : float, default 298.15
            Temperature in Kelvin
            
        Returns
        -------
        float
            Osmotic coefficient
        """
        
        if ionic_strength <= 0:
            return 1.0
        
        sqrt_I = np.sqrt(ionic_strength)
        
        # Debye-Hückel term
        phi_DH = -self.A_phi * sqrt_I / (1 + 1.2 * sqrt_I)
        
        # Pitzer virial terms
        phi_virial = 0.0
        
        # Binary terms
        cations = {k: v for k, v in composition.items() if self._get_charge(k) > 0}
        anions = {k: v for k, v in composition.items() if self._get_charge(k) < 0}
        
        for cation, m_c in cations.items():
            z_c = self._get_charge(cation)
            for anion, m_a in anions.items():
                z_a = self._get_charge(anion)
                
                params = self._get_binary_params(cation, anion)
                if params:
                    beta0, beta1, beta2, Cphi = params
                    
                    # B_phi and C_phi functions
                    alpha1 = 2.0
                    g_alpha1 = 2 * (1 - (1 + alpha1 * sqrt_I) * np.exp(-alpha1 * sqrt_I)) / (alpha1**2 * ionic_strength)
                    B_phi = beta0 + beta1 * np.exp(-alpha1 * sqrt_I)
                    
                    if beta2 != 0:
                        alpha2 = 50.0
                        B_phi += beta2 * np.exp(-alpha2 * sqrt_I)
                    
                    C_phi = Cphi / (2 * np.sqrt(abs(z_c * z_a)))
                    
                    phi_virial += m_c * m_a * (B_phi + m_c * m_a * C_phi)
        
        osmotic_coeff = 1 + phi_DH + phi_virial
        
        return osmotic_coeff
