def unfold(sequence: str, T: float = 298.15, P: float = 1.0) -> Dict[str, float]:
    """
    Calculate protein unfolding thermodynamics.
    
    Parameters
    ----------
    sequence : str
        Protein sequence
    T : float, default 298.15
        Temperature in Kelvin
    P : float, default 1.0
        Pressure in bar
        
    Returns
    -------
    dict
        Unfolding thermodynamics
    """
    
    folded = _protein_calculator.protein_properties(sequence, T, P)
    unfolded = _protein_calculator._calculate_unfolded_properties(sequence, T, P)
    
    delta_G = unfolded['G'] - folded['G']
    delta_H = unfolded['H'] - folded['H']  
    delta_S = unfolded['S'] - folded['S']
    
    # Melting temperature (where delta_G = 0)
    if delta_S != 0:
        T_m = delta_H / delta_S
    else:
        T_m = np.inf
    
    return {
        'delta_G': delta_G,
        'delta_H': delta_H,
        'delta_S': delta_S,
        'T_m': T_m,
        'K_unfold': np.exp(-delta_G / (8.314 * T)) if delta_G != 0 else 1.0,
        'fraction_unfolded': 1.0 / (1.0 + np.exp(-delta_G / (8.314 * T))) if delta_G != 0 else 0.5
    }

class ProteinThermodynamics:
    """
    Protein thermodynamics calculator.
    
    This class implements group contribution methods for calculating
    thermodynamic properties of proteins and biomolecules.
    """
    
    def __init__(self):
        """Initialize the protein thermodynamics calculator."""
        self.amino_acid_data = None
        self._load_amino_acid_data()
    
    def _load_amino_acid_data(self):
        """Load amino acid thermodynamic data."""
        try:
            # This would load from the protein.csv database
            # For now, create simplified data
            amino_acids = []
            for code, info in AMINO_ACIDS.items():
                amino_acids.append({
                    'code': code,
                    'name': info['name'],
                    'abbrev': info['abbrev'],
                    'formula': info['formula'],
                    # Placeholder thermodynamic data (would come from database)
                    'G': -100000 - np.random.random() * 50000,  # J/mol
                    'H': -150000 - np.random.random() * 50000,  # J/mol
                    'S': 100 + np.random.random() * 50,         # J/(mol·K)
                    'Cp': 150 + np.random.random() * 50         # J/(mol·K)
                })
            
            self.amino_acid_data = pd.DataFrame(amino_acids)
            
        except Exception as e:
            warnings.warn(f"Could not load amino acid data: {e}")
            self.amino_acid_data = pd.DataFrame()
    
    def protein_properties(self, sequence: str, 
                         T: float = 298.15, P: float = 1.0,
                         state: str = 'aq') -> Dict[str, float]:
        """
        Calculate thermodynamic properties of a protein from its sequence.
        
        Parameters
        ----------
        sequence : str
            Protein sequence in single-letter amino acid code
        T : float, default 298.15
            Temperature in Kelvin
        P : float, default 1.0
            Pressure in bar
        state : str, default 'aq'
            Physical state ('aq' for aqueous, 'cr' for crystalline)
            
        Returns
        -------
        dict
            Thermodynamic properties of the protein
        """
        
        if self.amino_acid_data.empty:
            raise ValueError("Amino acid data not available")
        
        # Count amino acid residues
        aa_counts = {}
        for aa in sequence:
            if aa.upper() in AMINO_ACIDS:
                aa_counts[aa.upper()] = aa_counts.get(aa.upper(), 0) + 1
            else:
                warnings.warn(f"Unknown amino acid: {aa}")
        
        # Calculate properties by group contribution
        total_G = 0
        total_H = 0
        total_S = 0
        total_Cp = 0
        
        for aa_code, count in aa_counts.items():
            aa_data = self.amino_acid_data[self.amino_acid_data['code'] == aa_code]
            if not aa_data.empty:
                aa_row = aa_data.iloc[0]
                total_G += count * aa_row['G']
                total_H += count * aa_row['H']
                total_S += count * aa_row['S']
                total_Cp += count * aa_row['Cp']
        
        # Apply peptide bond corrections
        n_peptide_bonds = len(sequence) - 1
        if n_peptide_bonds > 0:
            # Peptide bond formation: dehydration reaction
            # -H2O per peptide bond
            water_props = self._get_water_properties(T, P)
            total_G += n_peptide_bonds * water_props['G']  # Remove water
            total_H += n_peptide_bonds * water_props['H']
            total_S += n_peptide_bonds * water_props['S']
            total_Cp += n_peptide_bonds * water_props['Cp']
        
        # Temperature corrections (simplified)
        if T != 298.15:
            dT = T - 298.15
            total_G += total_H * (1 - T/298.15) + total_S * dT
            total_H += total_Cp * dT  # Assuming constant Cp
            total_S += total_Cp * np.log(T/298.15)
        
        return {
            'G': total_G,
            'H': total_H,
            'S': total_S,
            'Cp': total_Cp,
            'length': len(sequence),
            'composition': aa_counts,
            'molecular_weight': self._calculate_molecular_weight(aa_counts)
        }
    
    def protein_stability(self, sequence: str, 
                         T_range: np.ndarray,
                         P: float = 1.0) -> Dict[str, np.ndarray]:
        """
        Calculate protein stability as a function of temperature.
        
        Parameters
        ----------
        sequence : str
            Protein sequence
        T_range : array
            Temperature range in Kelvin
        P : float, default 1.0
            Pressure in bar
            
        Returns
        -------
        dict
            Temperature-dependent stability properties
        """
        
        results = {
            'T': T_range,
            'G_folded': np.zeros_like(T_range),
            'G_unfolded': np.zeros_like(T_range),
            'delta_G_fold': np.zeros_like(T_range),
            'fraction_folded': np.zeros_like(T_range)
        }
        
        # Calculate properties at each temperature
        for i, T in enumerate(T_range):
            # Folded state (compact)
            folded_props = self.protein_properties(sequence, T, P, state='aq')
            
            # Unfolded state (extended) - simplified as sum of individual residues
            unfolded_props = self._calculate_unfolded_properties(sequence, T, P)
            
            results['G_folded'][i] = folded_props['G']
            results['G_unfolded'][i] = unfolded_props['G']
            results['delta_G_fold'][i] = folded_props['G'] - unfolded_props['G']
            
            # Fraction folded (equilibrium)
            if results['delta_G_fold'][i] != 0:
                K_fold = np.exp(-results['delta_G_fold'][i] / (8.314 * T))
                results['fraction_folded'][i] = K_fold / (1 + K_fold)
            else:
                results['fraction_folded'][i] = 0.5
        
        return results
    
    def amino_acid_contribution(self, aa_code: str, 
                              T: float = 298.15, P: float = 1.0) -> Dict[str, float]:
        """
        Get thermodynamic contribution of a single amino acid.
        
        Parameters
        ----------
        aa_code : str
            Single-letter amino acid code
        T : float, default 298.15
            Temperature in Kelvin
        P : float, default 1.0
            Pressure in bar
            
        Returns
        -------
        dict
            Thermodynamic properties of the amino acid
        """
        
        if self.amino_acid_data.empty:
            raise ValueError("Amino acid data not available")
        
        aa_code = aa_code.upper()
        if aa_code not in AMINO_ACIDS:
            raise ValueError(f"Unknown amino acid: {aa_code}")
        
        aa_data = self.amino_acid_data[self.amino_acid_data['code'] == aa_code]
        if aa_data.empty:
            raise ValueError(f"No data available for amino acid: {aa_code}")
        
        aa_row = aa_data.iloc[0]
        
        # Temperature correction
        G_std = aa_row['G']
        H_std = aa_row['H']
        S_std = aa_row['S']
        Cp_std = aa_row['Cp']
        
        if T != 298.15:
            dT = T - 298.15
            G_T = G_std + H_std * (1 - T/298.15) + S_std * dT
            H_T = H_std + Cp_std * dT
            S_T = S_std + Cp_std * np.log(T/298.15)
        else:
            G_T, H_T, S_T = G_std, H_std, S_std
        
        return {
            'G': G_T,
            'H': H_T,
            'S': S_T,
            'Cp': Cp_std,
            'name': aa_row['name'],
            'formula': aa_row['formula']
        }
    
    def _get_water_properties(self, T: float, P: float) -> Dict[str, float]:
        """Get water thermodynamic properties."""
        try:
            from ..models import water_SUPCRT92
            # This would use the water model, but for now use constants
            return {
                'G': -237129,  # J/mol at 25°C
                'H': -285830,  # J/mol
                'S': 69.95,    # J/(mol·K)
                'Cp': 75.31    # J/(mol·K)
            }
        except:
            # Fallback values
            return {
                'G': -237129,
                'H': -285830,
                'S': 69.95,
                'Cp': 75.31
            }
    
    def _calculate_molecular_weight(self, aa_counts: Dict[str, int]) -> float:
        """Calculate molecular weight from amino acid composition."""
        
        # Approximate molecular weights (g/mol)
        mw_data = {
            'A': 89.1, 'R': 174.2, 'N': 132.1, 'D': 133.1, 'C': 121.0,
            'Q': 146.1, 'E': 147.1, 'G': 75.1, 'H': 155.2, 'I': 131.2,
            'L': 131.2, 'K': 146.2, 'M': 149.2, 'F': 165.2, 'P': 115.1,
            'S': 105.1, 'T': 119.1, 'W': 204.2, 'Y': 181.2, 'V': 117.1
        }
        
        total_mw = 0
        for aa, count in aa_counts.items():
            total_mw += count * mw_data.get(aa, 110.0)  # Average if unknown
        
        # Subtract water molecules for peptide bonds
        n_peptide_bonds = sum(aa_counts.values()) - 1
        total_mw -= n_peptide_bonds * 18.015  # H2O
        
        return total_mw
    
    def _calculate_unfolded_properties(self, sequence: str, T: float, P: float) -> Dict[str, float]:
        """Calculate properties of unfolded protein state."""
        
        # Simplified: unfolded state has higher entropy and enthalpy
        folded_props = self.protein_properties(sequence, T, P)
        
        # Unfolding typically increases entropy and enthalpy
        delta_H_unfold = len(sequence) * 5000  # ~5 kJ/mol per residue
        delta_S_unfold = len(sequence) * 20    # ~20 J/(mol·K) per residue
        
        return {
            'G': folded_props['G'] + delta_H_unfold - T * delta_S_unfold,
            'H': folded_props['H'] + delta_H_unfold,
            'S': folded_props['S'] + delta_S_unfold,
            'Cp': folded_props['Cp'] * 1.2  # Slightly higher Cp
        }
